206XMFLOAT4X4 SceneNode::getWorldTransform() const
207{
208 XMFLOAT4X4 transform = MathHelper::Identity4x4();
209 XMMATRIX T = XMLoadFloat4x4(&transform);
210
211 for (const SceneNode* node = this; node != nullptr; node = node->mParent)
212 {
213 XMMATRIX Tp = XMLoadFloat4x4(&node->getTransform());
214 T = Tp * T;
215 }
216 XMStoreFloat4x4(&transform, T);
