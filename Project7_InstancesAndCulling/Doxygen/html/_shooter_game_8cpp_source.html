470void Game::BuildRootSignature()
471{
472 CD3DX12_DESCRIPTOR_RANGE texTable;
473 texTable.Init(D3D12_DESCRIPTOR_RANGE_TYPE_SRV, 1, 0);
474
475 // Root parameter can be a table, root descriptor or root constants.
476 CD3DX12_ROOT_PARAMETER slotRootParameter[4];
477
478 // Perfomance TIP: Order from most frequent to least frequent.
479 slotRootParameter[0].InitAsDescriptorTable(1, &texTable, D3D12_SHADER_VISIBILITY_PIXEL);
480 slotRootParameter[1].InitAsConstantBufferView(0);
481 slotRootParameter[2].InitAsConstantBufferView(1);
482 slotRootParameter[3].InitAsConstantBufferView(2);
483
484 auto staticSamplers = GetStaticSamplers();
485
486 // A root signature is an array of root parameters.
487 //The Init function of the CD3DX12_ROOT_SIGNATURE_DESC class has two parameters that allow you to
488 //define an array of so - called static samplers your application can use.
489 CD3DX12_ROOT_SIGNATURE_DESC rootSigDesc(4, slotRootParameter,
490 (UINT)staticSamplers.size(), staticSamplers.data(), //6 samplers!
491 D3D12_ROOT_SIGNATURE_FLAG_ALLOW_INPUT_ASSEMBLER_INPUT_LAYOUT);
492
493 // create a root signature with a single slot which points to a descriptor range consisting of a single constant buffer
494 ComPtr<ID3DBlob> serializedRootSig = nullptr;
495 ComPtr<ID3DBlob> errorBlob = nullptr;
496 HRESULT hr = D3D12SerializeRootSignature(&rootSigDesc, D3D_ROOT_SIGNATURE_VERSION_1,
497 serializedRootSig.GetAddressOf(), errorBlob.GetAddressOf());
498
499 if (errorBlob != nullptr)
500 {
501 ::OutputDebugStringA((char*)errorBlob->GetBufferPointer());
502 }
503 ThrowIfFailed(hr);
504
505 ThrowIfFailed(md3dDevice->CreateRootSignature(
506 0,
507 serializedRootSig->GetBufferPointer(),
508 serializedRootSig->GetBufferSize(),
509 IID_PPV_ARGS(mRootSignature.GetAddressOf())));
510}

522 srvHeapDesc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_SHADER_VISIBLE;
523 ThrowIfFailed(md3dDevice->CreateDescriptorHeap(&srvHeapDesc, IID_PPV_ARGS(&mSrvDescriptorHeap)));
524
525 //
526 // Fill out the heap with actual descriptors.
527 //
528 CD3DX12_CPU_DESCRIPTOR_HANDLE hDescriptor(mSrvDescriptorHeap->GetCPUDescriptorHandleForHeapStart());
529

540
541 //This mapping enables the shader resource view (SRV) to choose how memory gets routed to the 4 return components in a shader after a memory fetch.
542 //When a texture is sampled in a shader, it will return a vector of the texture data at the specified texture coordinates.
543 //This field provides a way to reorder the vector components returned when sampling the texture.

816void Game::DrawRenderItems(ID3D12GraphicsCommandList* cmdList, const std::vector<RenderItem*>& ritems)
817{
818 UINT objCBByteSize = d3dUtil::CalcConstantBufferByteSize(sizeof(ObjectConstants));
819 UINT matCBByteSize = d3dUtil::CalcConstantBufferByteSize(sizeof(MaterialConstants));
820
821 auto objectCB = mCurrFrameResource->ObjectCB->Resource();
822 auto matCB = mCurrFrameResource->MaterialCB->Resource();
823
824 // For each render item...
825 for (size_t i = 0; i < ritems.size(); ++i)
826 {
827 auto ri = ritems[i];
828
829 cmdList->IASetVertexBuffers(0, 1, &ri->Geo->VertexBufferView());
830 cmdList->IASetIndexBuffer(&ri->Geo->IndexBufferView());
831 cmdList->IASetPrimitiveTopology(ri->PrimitiveType);
832
833 //step18
834 CD3DX12_GPU_DESCRIPTOR_HANDLE tex(mSrvDescriptorHeap->GetGPUDescriptorHandleForHeapStart());
835 tex.Offset(ri->Mat->DiffuseSrvHeapIndex, mCbvSrvDescriptorSize);
836
837 D3D12_GPU_VIRTUAL_ADDRESS objCBAddress = objectCB->GetGPUVirtualAddress() + ri->ObjCBIndex * objCBByteSize;
838 D3D12_GPU_VIRTUAL_ADDRESS matCBAddress = matCB->GetGPUVirtualAddress() + ri->Mat->MatCBIndex * matCBByteSize;
839
840 //step19
841 cmdList->SetGraphicsRootDescriptorTable(0, tex);
842 cmdList->SetGraphicsRootConstantBufferView(1, objCBAddress);
843 cmdList->SetGraphicsRootConstantBufferView(3, matCBAddress);
844
845 cmdList->DrawIndexedInstanced(ri->IndexCount, 1, ri->StartIndexLocation, ri->BaseVertexLocation, 0);
846 }
